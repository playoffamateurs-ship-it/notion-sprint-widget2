<script>

  // ====== PARAMS ======

  const SPRINT1_START = "2026-01-05"; // YYYY-MM-DD

  const SPRINT_LENGTH_DAYS = 14;

  // ====== Helpers ======

  const pad = (n) => String(n).padStart(2, "0");

  const formatFR = (d) => `${pad(d.getDate())}/${pad(d.getMonth()+1)}/${d.getFullYear()}`;

  const startOfDay = (d) => new Date(d.getFullYear(), d.getMonth(), d.getDate());

  const addDays = (d, days) => { const x = new Date(d); x.setDate(x.getDate()+days); return x; };

  const daysBetween = (a,b) => Math.floor((b-a)/(1000*60*60*24));

  const clamp = (v,min,max) => Math.max(min, Math.min(max, v));

  function sprintBounds(sprint1Start, sprintIndex){

    const start = addDays(sprint1Start, (sprintIndex-1)*SPRINT_LENGTH_DAYS);

    const end = addDays(start, SPRINT_LENGTH_DAYS-1);

    return { start, end };

  }

  function compute(){

    const today = startOfDay(new Date());

    const sprint1 = startOfDay(new Date(SPRINT1_START + "T00:00:00"));

    const deltaDays = daysBetween(sprint1, today);

    let sprintIndex = Math.floor(deltaDays / SPRINT_LENGTH_DAYS) + 1;

    if (deltaDays < 0) sprintIndex = 1;

    const cur = sprintBounds(sprint1, sprintIndex);

    const prev = sprintBounds(sprint1, Math.max(1, sprintIndex-1));

    const next = sprintBounds(sprint1, sprintIndex+1);

    const dayInSprint = clamp(daysBetween(cur.start, today)+1, 1, SPRINT_LENGTH_DAYS);

    const progress = clamp((dayInSprint-1)/(SPRINT_LENGTH_DAYS-1), 0, 1);

    const pct = Math.round(progress*100);

    const isLastDay = dayInSprint === SPRINT_LENGTH_DAYS;

    const pill = isLastDay ? "Dernier jour" : `J+${dayInSprint-1} / ${SPRINT_LENGTH_DAYS}`;

    // Header

    document.getElementById("currentTitle").textContent = `Sprint ${sprintIndex}`;

    document.getElementById("pill").textContent = pill;

    document.getElementById("currentRange").textContent =

      `Du ${formatFR(cur.start)} au ${formatFR(cur.end)} (14 jours)`;

    document.getElementById("pct").textContent = `${pct}%`;

    document.getElementById("kpi").textContent = isLastDay

      ? "Fin de sprint aujourd’hui"

      : `Aujourd’hui : ${formatFR(today)}`;

    // Segments labels

    document.getElementById("prevName").textContent = `Sprint ${Math.max(1, sprintIndex-1)}`;

    document.getElementById("curName").textContent = `Sprint ${sprintIndex}`;

    document.getElementById("nextName").textContent = `Sprint ${sprintIndex+1}`;

    document.getElementById("prevDates").textContent = `Du ${formatFR(prev.start)} au ${formatFR(prev.end)}`;

    document.getElementById("curDates").textContent  = `Du ${formatFR(cur.start)} au ${formatFR(cur.end)}`;

    document.getElementById("nextDates").textContent = `Du ${formatFR(next.start)} au ${formatFR(next.end)}`;

    // Progress bar in current segment

    const miniFill = document.getElementById("miniFill");

    miniFill.style.width = `${pct}%`;

    // Today marker position across the 3 segments:

    // - If today is in previous sprint -> place within first third

    // - If in current sprint -> middle third

    // - If in next sprint -> last third

    // We'll compute a normalized position 0..1.

    let pos = 0.5;

    const inPrev = today >= prev.start && today <= prev.end;

    const inCur  = today >= cur.start  && today <= cur.end;

    const inNext = today >= next.start && today <= next.end;

    const segFrac = (d, segStart) => clamp(daysBetween(segStart, d)/(SPRINT_LENGTH_DAYS-1), 0, 1);

    if (inPrev){

      pos = (0/3) + (segFrac(today, prev.start) / 3);

    } else if (inCur){

      pos = (1/3) + (segFrac(today, cur.start)  / 3);

    } else if (inNext){

      pos = (2/3) + (segFrac(today, next.start) / 3);

    } else if (today < prev.start){

      pos = 0; // avant la fenêtre

    } else {

      pos = 1; // après la fenêtre

    }

    const marker = document.getElementById("marker");

    marker.style.left = `${pos * 100}%`;

    document.getElementById("todayLabel").textContent = `Today • ${formatFR(today)}`;

  }

  compute();

  setInterval(compute, 10 * 60 * 1000);

</script>
